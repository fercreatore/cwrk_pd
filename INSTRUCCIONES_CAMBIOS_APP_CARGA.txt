CAMBIO 3: app_carga.py — Nueva funcion _detectar_proveedor_por_articulos

Agregar ANTES de la funcion _procesar_excel_auto():

def _detectar_proveedor_por_articulos(codigos: list) -> dict | None:
    """
    Dado una lista de codigos de articulo (modelo/referencia del Excel),
    busca en la BD cual es el proveedor que tiene esos articulos.
    Retorna dict con config del proveedor o None.
    """
    if not codigos:
        return None
    try:
        muestra = codigos[:10]
        placeholders = ",".join(["?"] * len(muestra))
        sql = f"""
            SELECT TOP 1 a.proveedor, COUNT(*) as hits
            FROM msgestion01art.dbo.articulo a
            WHERE a.descripcion_1 IN ({placeholders})
               OR CAST(a.codigo AS VARCHAR) IN ({placeholders})
            GROUP BY a.proveedor
            ORDER BY hits DESC
        """
        params = muestra + muestra
        from proveedores_db import _get_conn, obtener_pricing_proveedor, listar_proveedores_activos
        with _get_conn() as conn:
            cursor = conn.cursor()
            cursor.execute(sql, params)
            row = cursor.fetchone()
            if row and row.proveedor:
                provs = listar_proveedores_activos()
                prov_base = provs.get(row.proveedor)
                if prov_base:
                    prov = dict(prov_base)
                    pricing = obtener_pricing_proveedor(row.proveedor)
                    prov.update(pricing)
                    prov["_fuente"] = "articulos BD"
                    return prov
    except Exception as e:
        import logging
        logging.getLogger("app_carga").warning(f"Error detectando proveedor por articulos: {e}")
    return None
